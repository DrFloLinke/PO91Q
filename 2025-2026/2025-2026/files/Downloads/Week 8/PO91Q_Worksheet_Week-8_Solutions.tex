\begin{align*}\label{eq:beta1est}
\hat{\beta_{0}} &=  \bar{Y} - \hat{\beta_{1}}\bar{X} \\[0.25ex]
                &= 1495 - 39.69 \times 39 \\
                &= -53.09
\end{align*}

\begin{align*}
\hat{\beta_{1}} &= \dfrac{\Sigma_{i=1}^{N} (X_{i}-\bar{X})(Y_{i}-\bar{Y})}{\Sigma_{i=1}^{N} (X_{i} - \bar{X})^{2}} \\[0.25ex]
                &= \frac{83200}{2096} \\
                &= 39.69
\end{align*}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}

\item
  Specify the SRF and interpret the estimators of \(\beta_{0}\) and
  \(\beta_{1}\).
\end{enumerate}

\begin{equation*}
\widehat{\text{income}} = -53.09 + 39.7 \; \text{age}
\end{equation*}

\begin{itemize}
\tightlist
